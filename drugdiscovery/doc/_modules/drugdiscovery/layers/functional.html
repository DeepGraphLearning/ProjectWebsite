[docs]def multinomial(input, num_samples, replacement=False):
    """
    Fast multinomial sampling. This is the default implementation in PyTorch v1.6.0+.

    Parameters:
        input (Tensor): unnormalized distribution
        num_samples (int): number of samples
        replacement (bool, optional): sample with replacement or not
    """
    if replacement:
        return torch.multinomial(input, num_samples, replacement)

    rand = torch.rand_like(input).log() / input
    samples = rand.topk(num_samples).indices
    return samples




[docs]def shifted_softplus(input):
    """
    Shifted softplus function.

    Parameters:
        input (Tensor): input tensor
    """
    return F.softplus(input) - F.softplus(torch.zeros(1, device=input.device))




[docs]def multi_slice_mask(starts, ends, length):
    """
    Generate a mask from multiple non-overlapping slices.

    Example::

        >>> mask = F.multi_slice_mask(torch.tensor([0, 3, 6]), torch.tensor([2, 5, 8]), 9)
        >>> assert (mask == torch.tensor([1, 1, 0, 1, 1, 0, 1, 1, 0])).all()

    Parameters:
        starts (Tensor): start indexes of slices
        ends (Tensor): end indexes of slices
        length (int): length of mask
    """
    mask = torch.zeros(length, dtype=torch.long, device=starts.device)
    mask[starts] += 1
    mask[ends] -= 1
    mask = mask.cumsum(0).bool()
    return mask
