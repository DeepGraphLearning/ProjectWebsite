    def __init__(self, node_model, edge_model, num_node_sample=10, num_edge_sample=50):
        super(AutoregressiveGeneration, self).__init__()
        self.node_model = node_model
        self.edge_model = edge_model
        self.num_node_sample = num_node_sample
        self.num_edge_sample = num_edge_sample

    def forward(self, batch):
        """"""
        all_loss = torch.tensor(0, dtype=torch.float32, device=self.device)
        metric = {}

        graph, target = self.sample_node(batch["graph"])
        log_likelihood = self.node_model(graph, target, all_loss, metric)
        log_likelihood = log_likelihood.mean()
        metric["node log likelihood"] = log_likelihood
        all_loss += -log_likelihood

        graph, target = self.sample_edge(batch["graph"])
        log_likelihood = self.edge_model(graph, target, all_loss, metric)
        log_likelihood = log_likelihood.mean()
        metric["edge log likelihood"] = log_likelihood
        all_loss += -log_likelihood

        return all_loss, metric

    def evaluate(self, batch):
        pred = None
        metric = {}

        graph, target, valid = self.all_node(batch["graph"])
        log_likelihood = self.node_model(graph, target)
        log_likelihood = torch.sum(log_likelihood * valid) / torch.sum(valid)
        metric["node log likelihood"] = log_likelihood

        graph, target, valid = self.all_edge(batch["graph"])
        log_likelihood = self.edge_model(graph, target)
        log_likelihood = torch.sum(log_likelihood * valid) / torch.sum(valid)
        metric["edge log likelihood"] = log_likelihood

        return pred, metric

    @torch.no_grad()
    def sample_node(self, graph):
        graph = graph.repeat(self.num_node_sample)
        num_nodes = graph.num_nodes
        num_keep_nodes = torch.rand(len(graph), device=graph.device) * (num_nodes - 1)
        num_keep_nodes = num_keep_nodes.long()

        starts = graph.num_cum_nodes - graph.num_nodes
        ends = starts + num_keep_nodes
        mask = functional.multi_slice_mask(starts, ends, graph.num_node)

        new_graph = graph.subgraph(mask)
        target = graph.subgraph(ends).atom_type
        return new_graph, target

    def all_range_mask(self, num_xs, num_cum_xs, lengths=None):
        # extract a mask that concates the following n*num_repeat masks
        # graphs[0]: [1, 0, ...]
        # ...
        # graphs[-1]: [1, 0, ...]
        # graphs[0]: [1, 1, 0, ...]
        # ...
        # graphs[-1]: [1, 1, 0, ...]
        # ...

        starts = num_cum_xs - num_xs
        if lengths is None:
            num_max_x = num_xs.max().item()
            lengths = torch.arange(num_max_x)
        lengths, starts = torch.meshgrid(lengths, starts)
        lengths = torch.min(lengths, num_xs.unsqueeze(0))
        valid = lengths <= num_xs.unsqueeze(0)
        ends = starts + lengths

        starts = starts.flatten()
        ends = ends.flatten()
        valid = valid.flatten()
        return starts, ends, valid

    @torch.no_grad()

        new_graph = graph.subgraph(mask)
        target = graph.subgraph(ends).atom_type

    @torch.no_grad()
    def sample_edge(self, graph):
        graph = graph.repeat(self.num_edge_sample)
        # assume already BFS ordered
        # QM9: max_edge_unroll: 5 ~ 6
        max_edge_unroll = (graph.edge_list[:, 0] - graph.edge_list[:, 1]).abs().max().item()
        num_nodes = graph.num_nodes
        num_sqrt_edges = torch.rand(len(graph), device=graph.device) * num_nodes + 1
        num_sqrt_edges = num_sqrt_edges.long()
        num_keep_edges = num_sqrt_edges ** 2 - \
                         torch.rand(len(graph), device=graph.device) * num_sqrt_edges.clamp(max=max_edge_unroll + 1)
        num_keep_edges = num_keep_edges.int()

        # use large / small to give a unique id to an **undirected** edge
        node_out = graph.edge_list[:, 0] - graph._offsets
        node_in = graph.edge_list[:, 1] - graph._offsets
        edge_id = node_out * graph.num_nodes[graph.edge2graph] + node_in
        rev_edge_id = node_in * graph.num_nodes[graph.edge2graph] + node_out
        undirected_edge_id = torch.max(edge_id, rev_edge_id)

        mask = undirected_edge_id < num_keep_edges[graph.edge2graph]

        positive_edge = edge_id == num_keep_edges[graph.edge2graph]
        positive_graph = scatter_add(positive_edge.long(), graph.edge2graph, dim=0, dim_size=graph.batch_size).bool()
        target = - torch.ones(graph.batch_size, dtype=torch.long, device=graph.device)
        target[positive_graph] = graph.bond_type[positive_edge]

        new_graph = graph.edge_mask(mask)
        return new_graph, target

    @torch.no_grad()
    def all_edge(self, graph):
        num_max_edge = graph.num_edges.max().item()
        max_edge_unroll = (graph.edge_list[:, 0] - graph.edge_list[:, 1]).abs().max().item()
        lengths = torch.arange(num_max_edge)
        num_edge_unrolls = lengths.sqrt().ceil().long() ** 2 - lengths
        lengths = lengths[num_edge_unrolls <= max_edge_unroll]

        starts, ends, valid = self.all_range_mask(graph.num_edges, graph.num_cum_edges, lengths)

        num_keep_edges = ends - starts
        num_repeat = len(starts) / len(graph)
        graph = graph.repeat(num_repeat)

        # use large / small to give a unique id to an **undirected** edge
        node_out = graph.edge_list[:, 0] - graph._offsets
        node_in = graph.edge_list[:, 1] - graph._offsets
        edge_id = node_out * graph.num_nodes[graph.edge2graph] + node_in
        rev_edge_id = node_in * graph.num_nodes[graph.edge2graph] + node_out
        undirected_edge_id = torch.max(edge_id, rev_edge_id)

        mask = undirected_edge_id < num_keep_edges[graph.edge2graph]

        positive_edge = edge_id == num_keep_edges[graph.edge2graph]
        positive_graph = scatter_add(positive_edge.long(), graph.edge2graph, dim=0, dim_size=graph.batch_size).bool()
        target = - torch.ones(graph.batch_size, dtype=torch.long, device=graph.device)
        target[positive_graph] = graph.bond_type[positive_edge]

        new_graph = graph.edge_mask(mask)
        return new_graph, target, valid
