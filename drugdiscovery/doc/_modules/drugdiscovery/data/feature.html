@R.register("features.atom.chirality")
def atom_chirality(atom):
    atom_dype = [atom.GetAtomicNum() - 1]
    chiral_type = [[
        Chem.rdchem.ChiralType.CHI_UNSPECIFIED,
        Chem.rdchem.ChiralType.CHI_TETRAHEDRAL_CW,
        Chem.rdchem.ChiralType.CHI_TETRAHEDRAL_CCW,
        Chem.rdchem.ChiralType.CHI_OTHER
    ].index(atom.GetChiralTag())]
    return atom_dype + chiral_type



[docs]@R.register("features.atom.position")
def atom_position(atom):
    """
    Node position.
    Return 3D position if available, otherwise 2D position is returned.
    """
    mol = atom.GetOwningMol()
    if mol.GetNumConformers() == 0:
        mol.Compute2DCoords()
    conformer = mol.GetConformer()
    pos = conformer.GetAtomPosition(atom.GetIdx())
    return list(pos)

[docs]@R.register("features.bond.property_prediction")
def property_prediction_edge_feature(bond):
    """Property prediction edge feature."""
    bt = bond.GetBondType()
    return [
        bt == Chem.rdchem.BondType.SINGLE,
        bt == Chem.rdchem.BondType.DOUBLE,
        bt == Chem.rdchem.BondType.TRIPLE,
        bt == Chem.rdchem.BondType.AROMATIC,
        bond.GetIsConjugated(),
        bond.IsInRing()
    ]



@R.register("features.bond.direction")
def bond_direction(bond):
    bond_type = [[
        Chem.rdchem.BondType.SINGLE,
        Chem.rdchem.BondType.DOUBLE,
        Chem.rdchem.BondType.TRIPLE,
        Chem.rdchem.BondType.AROMATIC,
    ].index(bond.GetBondType())]
    bond_dir = [[
        Chem.rdchem.BondDir.NONE,
        Chem.rdchem.BondDir.ENDUPRIGHT,
        Chem.rdchem.BondDir.ENDDOWNRIGHT
    ].index(bond.GetBondDir())]
    return bond_type + bond_dir

def ECFP(mol, radius=2, length=1024):
    """Extended Connectivity Fingerprint graph feature."""
    ecfp = AllChem.GetMorganFingerprintAsBitVect(mol, radius, length)
    return list(ecfp)

def one_of_k_encoding(x, allowable_set):
    if x not in allowable_set:
        raise Exception("input {0} not in allowable set{1}:".format(x, allowable_set))
    return list(map(lambda s: 1 if x == s else 0, allowable_set))


def one_of_k_encoding_unk(x, allowable_set):
    if x not in allowable_set:
        # logger.info('Unknown detected: {}'.format(x))
        x = allowable_set[-1]
    return list(map(lambda s: 1 if x == s else 0, allowable_set))
