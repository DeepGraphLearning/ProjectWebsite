    def get_center(self, mol_re, mol_prod):
        re_num2id = {}
        re_id2num = {}
        for atom in mol_re.GetAtoms():
            map_num = atom.GetAtomMapNum()
            if map_num:
                re_num2id[map_num] = atom.GetIdx()
                re_id2num[atom.GetIdx()] = map_num             
        prod_num2id = {}
        prod_id2num = {}
        for atom in mol_prod.GetAtoms():
            map_num = atom.GetAtomMapNum()
            if map_num:
                prod_num2id[map_num] = atom.GetIdx()
                prod_id2num[atom.GetIdx()] = map_num
        centers = []
        mapped_centers = []
        all_prod_bonds = mol_prod.GetBonds()
        for bond_ in all_prod_bonds:
            type_ = bond_.GetBondType()
            begin_id = bond_.GetBeginAtomIdx()
            end_id = bond_.GetEndAtomIdx()
            begin_num = prod_id2num[begin_id]
            end_num = prod_id2num[end_id]
            re_bond = mol_re.GetBondBetweenAtoms(re_num2id[begin_num], re_num2id[end_num])
            if re_bond is None:
                centers.append([begin_id, end_id])
                mapped_centers.append([begin_num, end_num])
        return centers, mapped_centers

    @property
    def atom_types(self):
        if not hasattr(self, "_atom_types"):
            atom_types = set()
            for graph in self.data:
                atom_types.update(graph.atom_type.tolist())
            self._atom_types = sorted(atom_types)
        return self._atom_types
